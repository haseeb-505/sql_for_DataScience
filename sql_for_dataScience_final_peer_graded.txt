iii. Are you able to infer anything from the location data provided between these two groups? Explain.
	Location data is not much useful for the city "Chandler" and Category "shopping" as there is not that much variety in the location. 
	There are only two locations and each of them with 5 and 7 review_count with same rating 4-5 stars.
SQL code used for analysis:

select name, city, review_count, 
        postal_code, hours, 
        ord_hour, star_rating
from    (select b.id,
            b.name, 
            b.city, 
            b.review_count, 
            b.postal_code,
            b.stars, 
            h.hours,
            CASE
                WHEN hours like "%monday%" THEN 1
                WHEN hours like "%tuesday%" THEN 2
                WHEN hours like "%wednesday%" THEN 3
                WHEN hours like "%thursday%" THEN 4
                WHEN hours like "%friday%" THEN 5
                WHEN hours like "%saturday%" THEN 6
                WHEN hours like "%sunday%" THEN 7
            END as ord_hour,
            CASE
                when b.stars between 2 and 3 then "2-3 stars"
                when b.stars between 4 and 5 then "4-5 stars"
            end as star_rating
    from business as b
    inner join hours as h on b.id=h.business_id) as a
inner join category as c on c.business_id=a.id
-- selecing the city and category and stars
where(
    (a.City == "Chandler" and c.category like "shopping")
    and 
    (a.stars between 2 and 3 
    or 
    a.stars between 4 and 5)
    )
 group by stars, ord_hour
 order by ord_hour, star_rating asc

SELECT 
			   B.name,
			   B.address,
			   B.city,
			   B.state,
			   B.postal_code,
			   B.review_count,
			   B.stars,
			   MAX(CASE
			   WHEN H.hours LIKE "%monday%" THEN TRIM(H.hours,'%MondayTuesWednesThursFriSatSun|%') 
			   END) AS monday_hours,
			   MAX(CASE
			   WHEN H.hours LIKE "%tuesday%" THEN TRIM(H.hours,'%MondayTuesWednesThursFriSatSun|%') 
			   END) AS tuesday_hours,
			   MAX(CASE
			   WHEN H.hours LIKE "%wednesday%" THEN TRIM(H.hours,'%MondayTuesWednesThursFriSatSun|%') 
			   END) AS wednesday_hours,
			   MAX(CASE
			   WHEN H.hours LIKE "%thursday%" THEN TRIM(H.hours,'%MondayTuesWednesThursFriSatSun|%') 
			   END) AS thursday_hours,
			   MAX(CASE
			   WHEN H.hours LIKE "%friday%" THEN TRIM(H.hours,'%MondayTuesWednesThursFriSatSun|%') 
			   END) AS friday_hours,
			   MAX(CASE
			   WHEN H.hours LIKE "%saturday%" THEN TRIM(H.hours,'%MondayTuesWednesThursFriSatSun|%') 
			   END) AS saturday_hours,
			   MAX(CASE
			   WHEN H.hours LIKE "%sunday%" THEN TRIM(H.hours,'%MondayTuesWednesThursFriSatSun|%') 
			   END) AS sunday_hours,
			   GROUP_CONCAT(DISTINCT(C.category)) AS categories,
			   GROUP_CONCAT(DISTINCT(A.name)) AS attributes,
			   B.is_open
		FROM business B
		INNER JOIN hours H
		ON B.id = H.business_id
		INNER JOIN category C
		ON B.id = C.business_id
		INNER JOIN attribute A
		ON B.id = A.business_id
		GROUP BY B.id
